function avg(values) {
  if (!values.length) return 0;
  return values.reduce((a, b) => a + Number(b || 0), 0) / values.length;
}

function num(value) {
  if (value === "" || value === null || value === undefined) return 0;
  const n = Number(value);
  return Number.isFinite(n) ? n : 0;
}

function displayTime(ts) {
  if (!ts) return "—";
  try {
    return new Date(ts).toLocaleTimeString([], {
      hour: "2-digit",
      minute: "2-digit",
    });
  } catch {
    return "—";
  }
}

function getBlofinUrl(pair) {
  if (!pair) return "https://blofin.com";
  return `https://blofin.com/futures/en/${pair.replace("/", "-")}`;
}

function getTradingViewSymbol(pair) {
  const compact = (pair || "BTC/USDT").replace("/", "");
  return `BLOFIN:${compact}`;
}

function getTradingViewInterval(timeframe) {
  const map = {
    "1m": "1",
    "3m": "3",
    "5m": "5",
    "15m": "15",
    "1h": "60",
    "4h": "240",
    "1d": "D",
  };
  return map[timeframe] || "3";
}

function calcRiskAmount(entry, stop, riskUnit = 1) {
  const e = num(entry);
  const s = num(stop);
  if (!e || !s) return 0;
  return Math.abs(e - s) * riskUnit;
}

function calcRR(entry, stop, target) {
  const e = num(entry);
  const s = num(stop);
  const t = num(target);
  const risk = Math.abs(e - s);
  const reward = Math.abs(t - e);
  if (!risk || !reward) return 0;
  return reward / risk;
}

function getPropMetrics(propAccount) {
  const dailyLossLimit = propAccount.accountSize * propAccount.dailyDrawdownPct;
  const maxDrawdownLimit = propAccount.accountSize * propAccount.maxDrawdownPct;
  const profitTarget = propAccount.accountSize * propAccount.profitTargetPct;

  const dailyLossUsed = Math.max(0, -propAccount.dailyRealizedPnl);
  const maxDrawdownUsed = Math.max(
    0,
    propAccount.startingBalance - propAccount.currentBalance
  );

  return {
    dailyLossLimit,
    maxDrawdownLimit,
    profitTarget,
    dailyLossUsed,
    maxDrawdownUsed,
    remainingDailyLoss: Math.max(0, dailyLossLimit - dailyLossUsed),
    remainingMaxDrawdown: Math.max(0, maxDrawdownLimit - maxDrawdownUsed),
  };
}

function evaluatePropCompliance(event, propAccount) {
  if (!event || propAccount.mode !== "prop") {
    return {
      qualified: true,
      notes: [],
      riskAmount: 0,
      rrToTp1: 0,
      rrToTp2: 0,
    };
  }

  const notes = [];
  const metrics = getPropMetrics(propAccount);

  const riskAmount = calcRiskAmount(event.entry, event.stop, 1);
  const rrToTp1 = calcRR(event.entry, event.stop, event.tp1);
  const rrToTp2 = calcRR(event.entry, event.stop, event.tp2);

  if (!event.entry || !event.stop) notes.push("Missing entry/stop data");
  if (riskAmount > propAccount.maxRiskPerTrade) notes.push("Risk exceeds max per trade");

  const bestRR = Math.max(rrToTp1 || 0, rrToTp2 || 0);
  if (bestRR && bestRR < propAccount.minRR) notes.push("RR below minimum");

  if (propAccount.lossesToday >= propAccount.maxLossesPerDay) {
    notes.push("Max losses for the day reached");
  }

  if (metrics.dailyLossUsed >= metrics.dailyLossLimit) {
    notes.push("Daily drawdown limit reached");
  }

  if (metrics.maxDrawdownUsed >= metrics.maxDrawdownLimit) {
    notes.push("Max overall drawdown reached");
  }

  if (
    propAccount.dailyProfitLock &&
    propAccount.dailyRealizedPnl >= propAccount.dailyProfitLock
  ) {
    notes.push("Daily profit lock reached");
  }

  return {
    qualified: notes.length === 0,
    notes,
    riskAmount,
    rrToTp1,
    rrToTp2,
  };
}

function safeReadStorage(key, fallback) {
  try {
    const raw = localStorage.getItem(key);
    if (!raw) return fallback;
    return JSON.parse(raw);
  } catch {
    return fallback;
  }
}

function eventKey(event) {
  return (
    event?.id ||
    `${event?.pair}_${event?.timestampUtc}_${event?.eventType}_${event?.sweepType}`
  );
}

              <span className="pill">

                {logMode === "event" ? selectedEvent?.session || "Waiting" : decisionForm.session}

            <a href={blofinUrl} target="_blank" rel="noreferrer" className="mini-link-btn">

            Keep the chart visible while you log so the pattern stays tied to the decision.

        </main>

        <aside className="panel logger-panel">
          <div className="panel-header">
            <div>
              <h2>{logMode === "event" ? "Trade Journal" : "Manual Trade Journal"}</h2>

                {logMode === "event" ? "Respond to selected radar event" : "Log without event card"}

              <span className="pill small">
                {logMode === "event" ? selectedEvent?.timeframe || "—" : decisionForm.timeframe}

              <span className="pill small">
                {logMode === "event" ? selectedEvent?.emaContext || "—" : decisionForm.emaContext}
              </span>
            </div>
          </div>

          {propAccount.mode === "prop" ? (
            <div className={selectedCompliance.qualified ? "compliance-box pass" : "compliance-box fail"}>

                <strong>
                  {selectedCompliance.qualified ? "Prop-Compliant Setup" : "Not Prop-Compliant"}

                <div className="form-grid two">

                      placeholder="SUI/USDT"

                      onChange={(e) => updateDecision("pair", e.target.value)}
                    />
                  </Field>

                  <Field label="Timeframe">
                    <select
                      value={decisionForm.timeframe || "3m"}
                      onChange={(e) => updateDecision("timeframe", e.target.value)}
                    >
                      <option value="1m">1m</option>
                      <option value="3m">3m</option>
                      <option value="5m">5m</option>
                      <option value="15m">15m</option>
                      <option value="1h">1h</option>
                    </select>
                  </Field>
                </div>

                <div className="form-grid two">
                  <Field label="Direction Bias">
                    <select
                      value={decisionForm.directionBias || "Short"}

                      <option value="Long">Long</option>
                      <option value="Short">Short</option>
                    </select>
                  </Field>

                  <Field label="Session">
                    <select
                      value={decisionForm.session || "New York"}

                      <option value="Asia">Asia</option>
                      <option value="London">London</option>
                      <option value="New York">New York</option>
                    </select>
                  </Field>
                </div>

                <div className="form-grid two">
                  <Field label="Event Type">

                      value={decisionForm.eventType || "SWEEP_DETECTED"}

                      <option value="SWEEP_ACCEPTED">SWEEP_ACCEPTED</option>

                  </Field>

                    <input
                      type="text"

              </>

              <Field label="Action Taken">
                <select
                  value={decisionForm.action || "Taken"}
                  onChange={(e) => updateDecision("action", e.target.value)}
                >
                  <option value="Taken">Taken</option>
                  <option value="Skipped">Skipped</option>
                  <option value="Missed">Missed</option>
                  <option value="Watched Only">Watched Only</option>
                </select>
              </Field>

              <Field label="Entry Timing">
                <select
                  value={decisionForm.timing || "On Confirmation"}
                  onChange={(e) => updateDecision("timing", e.target.value)}
                >
                  <option value="On Confirmation">On Confirmation</option>
                  <option value="Early">Early</option>
                  <option value="Late">Late</option>
                  <option value="Chase Entry">Chase Entry</option>
                </select>
              </Field>
            </div>

            <div className="form-grid two">
              <Field label="Plan Adherence">
                <select
                  value={decisionForm.planFollowed || "Yes"}
                  onChange={(e) => updateDecision("planFollowed", e.target.value)}
                >
                  <option value="Yes">Yes</option>
                  <option value="Mostly">Mostly</option>
                  <option value="No">No</option>
                </select>
              </Field>

              <Field label="Execution Mistake">
                <select
                  value={decisionForm.ruleBreak || "None"}
                  onChange={(e) => updateDecision("ruleBreak", e.target.value)}
                >
                  <option value="None">None</option>
                  <option value="Entered before confirmation">Entered before confirmation</option>
                  <option value="Chased move">Chased move</option>
                  <option value="Forced trade">Forced trade</option>
                  <option value="Ignored invalidation">Ignored invalidation</option>
                  <option value="Moved stop">Moved stop</option>
                  <option value="Closed early">Closed early</option>
                  <option value="Over leveraged">Over leveraged</option>
                  <option value="Revenge trade">Revenge trade</option>
                </select>
              </Field>
            </div>

            <div className="form-grid three">
              <Field label="Discipline Score">
                <input
                  type="number"
                  min="1"
                  max="10"
                  value={decisionForm.disciplineScore}
                  onChange={(e) => updateDecision("disciplineScore", e.target.value)}
                />
              </Field>

              <Field label="Setup Quality">
                <input
                  type="number"
                  min="1"
                  max="10"
                  value={decisionForm.setupQuality}
                  onChange={(e) => updateDecision("setupQuality", e.target.value)}
                />
              </Field>

              <Field label="Emotional Pressure">
                <input
                  type="number"
                  min="1"
                  max="10"
                  value={decisionForm.emotionalPressure}
                  onChange={(e) => updateDecision("emotionalPressure", e.target.value)}
                />
              </Field>
            </div>

            <div className="form-grid two">
              <Field label="Entry">
                <input
                  type="number"
                  step="any"
                  value={decisionForm.entry}
                  onChange={(e) => updateDecision("entry", e.target.value)}
                />
              </Field>

              <Field label="Stop">
                <input
                  type="number"
                  step="any"
                  value={decisionForm.stop}
                  onChange={(e) => updateDecision("stop", e.target.value)}
                />
              </Field>
            </div>

            <div className="form-grid three">
              <Field label="TP1">
                <input
                  type="number"

                  value={decisionForm.tp1}
                  onChange={(e) => updateDecision("tp1", e.target.value)}

              </Field>

              <Field label="TP2">
                <input
                  type="number"
                  step="any"

                  type="number"

            </div>

            <div className="form-grid one">
              <Field label="PnL">
                <input
                  type="number"
                  step="any"
                  value={decisionForm.pnl}
                  onChange={(e) => updateDecision("pnl", e.target.value)}
                />
              </Field>
            </div>

            <Field label="Decision Notes">
              <textarea
                rows="5"
                placeholder="Why did you take or skip it? What made it valid or invalid?"
                value={decisionForm.notes}
                onChange={(e) => updateDecision("notes", e.target.value)}
              />
            </Field>

            <Field label="Chart Screenshot">
              <input
                type="file"
                accept="image/*"
                onChange={(e) => {
                  const file = e.target.files?.[0];
                  if (!file) return;
                  const reader = new FileReader();
                  reader.onload = () => updateDecision("screenshot", reader.result);
                  reader.readAsDataURL(file);
                }}
              />
            </Field>

            {decisionForm.screenshot ? (
              <div className="screenshot-preview">
                <img src={decisionForm.screenshot} alt="Decision screenshot preview" />
              </div>
            ) : null}
          </div>

          <div className="button-row logger-actions">
            <button className="btn primary" onClick={saveDecision} type="button">
              Log Trade
            </button>
            <a href={blofinUrl} target="_blank" rel="noreferrer" className="btn secondary link-btn">
              Open on BloFin
            </a>
          </div>
        </aside>
      </section>

      <section className="bottom-section">
        <div className="panel">
          <div className="panel-header">
            <div>
              <h2>Recent Trade Logs</h2>
              <span className="panel-subtext">Saved event responses and manual journal entries</span>
            </div>
          </div>

          {loggedDecisions.length === 0 ? (
            <div className="empty-box">
              No logs yet. Select a radar event or switch to manual mode.
            </div>
          ) : (
            <div className="response-list horizontal">
              {loggedDecisions.map((item) => (

                      <MiniBox label="Entry" value={item.entry || "—"} />

                      <MiniBox label="TP2" value={item.tp2 || "—"} />

                        label="Risk"
                        value={item.riskAmount?.toFixed?.(2) || "—"}
                      />
                    </div>
                  ) : null}

                  {item.propQualified !== undefined ? (
                    <div className={item.propQualified ? "compliance-box pass" : "compliance-box fail"}>
                      <strong>{item.propQualified ? "Prop Qualified" : "Prop Rejected"}</strong>
                      {item.riskAmount ? (
                        <div className="mini-label">Risk: {Number(item.riskAmount).toFixed(2)}</div>
                      ) : null}
                      {item.complianceNotes?.length ? (
                        <ul className="compliance-list">
                          {item.complianceNotes.map((note) => (
                            <li key={note}>{note}</li>
                          ))}
                        </ul>
                      ) : null}
                    </div>
                  ) : null}

                  {item.notes ? <p className="response-notes">{item.notes}</p> : null}

                  {item.screenshot ? (
                    <div className="response-screenshot">
                      <img src={item.screenshot} alt={`${item.pair} screenshot`} />
                    </div>
                  ) : null}

                  <div className="event-actions">
                    <a
                      href={getBlofinUrl(item.pair)}
                      target="_blank"
                      rel="noreferrer"
                      className="mini-link-btn"
                    >
                      Open on BloFin
                    </a>
                  </div>
                </div>
              ))}
            </div>
          )}
        </div>
      </section>

      <section className="bottom-section">
        <div className="panel prop-panel compact">
          <div className="panel-header">
            <div>
              <h2>Prop Rules</h2>
              <span className="panel-subtext">Kept lower so the live workflow stays front and center</span>
            </div>

            <button
              className="mode-btn"
              onClick={() => setShowPropAdvanced((prev) => !prev)}
              type="button"
            >
              {showPropAdvanced ? "Hide Advanced" : "Show Advanced"}
            </button>
          </div>

          <div className="prop-compact-grid">
            <Field label="Mode">
              <select
                value={propAccount.mode}
                onChange={(e) => updateProp("mode", e.target.value)}
              >
                <option value="personal">Personal</option>
                <option value="prop">Prop</option>
              </select>
            </Field>

            <Field label="Account Size">
              <input
                type="number"
                value={propAccount.accountSize}
                onChange={(e) => updateProp("accountSize", e.target.value)}
              />
            </Field>

            <Field label="Max Risk / Trade">
              <input
                type="number"
                value={propAccount.maxRiskPerTrade}
                onChange={(e) => updateProp("maxRiskPerTrade", e.target.value)}
              />
            </Field>

            <Field label="Min RR">
              <input
                type="number"
                step="0.1"
                value={propAccount.minRR}
                onChange={(e) => updateProp("minRR", e.target.value)}
              />
            </Field>
          </div>

          <div className="prop-summary-strip">
            <div className="prop-summary-card">
              <span className="mini-label">Daily DD Used</span>
              <strong>{propMetrics.dailyLossUsed.toFixed(0)}</strong>
              <span className="prop-sub">
                / {propMetrics.dailyLossLimit.toFixed(0)}
              </span>
            </div>

            <div className="prop-summary-card">
              <span className="mini-label">Remaining Daily</span>
              <strong>{propMetrics.remainingDailyLoss.toFixed(0)}</strong>
              <span className="prop-sub">Loss room left</span>
            </div>

            <div className="prop-summary-card">
              <span className="mini-label">Losses Today</span>
              <strong>{propAccount.lossesToday}</strong>
              <span className="prop-sub">
                / {propAccount.maxLossesPerDay}
              </span>
            </div>
          </div>

          {showPropAdvanced ? (
            <div className="prop-advanced-wrap">
              <div className="form-grid two">
                <Field label="Firm">
                  <select
                    value={propAccount.firmName}
                    onChange={(e) => updateProp("firmName", e.target.value)}
                  >
                    <option value="Custom">Custom</option>
                    <option value="FTMO">FTMO</option>
                    <option value="FundedNext">FundedNext</option>
                    <option value="The5ers">The5ers</option>
                  </select>
                </Field>

                <Field label="Daily Profit Lock">
                  <input
                    type="number"
                    value={propAccount.dailyProfitLock}
                    onChange={(e) => updateProp("dailyProfitLock", e.target.value)}
                  />
                </Field>
              </div>

              <div className="form-grid three">
                <Field label="Profit Target %">
                  <input
                    type="number"
                    step="0.01"
                    value={propAccount.profitTargetPct}
                    onChange={(e) => updateProp("profitTargetPct", e.target.value)}
                  />
                </Field>

                <Field label="Daily DD %">
                  <input
                    type="number"
                    step="0.01"
                    value={propAccount.dailyDrawdownPct}
                    onChange={(e) => updateProp("dailyDrawdownPct", e.target.value)}
                  />
                </Field>

                <Field label="Max DD %">
                  <input
                    type="number"
                    step="0.01"
                    value={propAccount.maxDrawdownPct}
                    onChange={(e) => updateProp("maxDrawdownPct", e.target.value)}
                  />
                </Field>
              </div>

              <div className="form-grid three">
                <Field label="Max Losses / Day">
                  <input
                    type="number"
                    value={propAccount.maxLossesPerDay}
                    onChange={(e) => updateProp("maxLossesPerDay", e.target.value)}
                  />
                </Field>

                <Field label="Starting Balance">
                  <input
                    type="number"
                    value={propAccount.startingBalance}
                    onChange={(e) => updateProp("startingBalance", e.target.value)}
                  />
                </Field>

                <Field label="Current Balance">
                  <input
                    type="number"
                    value={propAccount.currentBalance}
                    onChange={(e) => updateProp("currentBalance", e.target.value)}
                  />
                </Field>
              </div>

              <div className="stats-grid compact-prop-stats">
                <StatCard
                  title="Profit Target"
                  value={propMetrics.profitTarget.toFixed(0)}
                  subtitle="Target dollars"
                />
                <StatCard
                  title="Daily DD Limit"
                  value={propMetrics.dailyLossLimit.toFixed(0)}
                  subtitle="Max daily loss"

                  title="Max DD Used"
                  value={propMetrics.maxDrawdownUsed.toFixed(0)}
                  subtitle="Overall drawdown"
                />
              </div>
            </div>
          ) : null}
        </div>
      </section>
    </div>
  );
}

function StatCard({ title, value, subtitle }) {
  return (
    <div className="stat-card">
      <div className="mini-label">{title}</div>
      <div className="stat-value">{value}</div>
      <div className="stat-sub">{subtitle}</div>
    </div>
  );
}

function Field({ label, children }) {
  return (
    <label className="field">
      <span>{label}</span>
      {children}
    </label>
  );
}

function MiniBox({ label, value }) {
  return (
    <div className="mini-box">
      <span className="mini-label">{label}</span>
      <strong>{value}</strong>
    </div>
  );
}
